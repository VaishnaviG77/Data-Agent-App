// ── SCREEN 2: Mapping & Cleaning (shows real column schema from uploaded files) ──
const MappingCleaning = ({ uploadedFiles }) => {
  const [selectedFileId, setSelectedFileId] = useState(null);
  const [fileDetails, setFileDetails]       = useState(null);
  const [loadingDetails, setLoadingDetails] = useState(false);

  // Universal schema fields (the target we want to map to)
  const universalSchema = [
    "transaction_id", "transaction_date", "sku_id", "product_name",
    "brand", "sku_category", "manufacturer_name", "region",
    "channel_type", "units_sold", "mrp", "invoice_value_incl_gst",
  ];

  // Load file details when user picks a file
  useEffect(() => {
    if (!selectedFileId) {
      setFileDetails(null);
      return;
    }
    setLoadingDetails(true);
    fetch(`${API_BASE}/files/${selectedFileId}`)
      .then(r => r.json())
      .then(data => {
        setFileDetails(data);
        setLoadingDetails(false);
      })
      .catch(() => setLoadingDetails(false));
  }, [selectedFileId]);

  const rawColumns = fileDetails?.col_names_list || [];

  // Simple keyword-based auto-mapping for display
  const autoMap = (rawCol) => {
    const c = rawCol.toLowerCase().replace(/[^a-z0-9]/g, " ").trim();
    if (c.includes("order id") || c.includes("orderid") || c.includes("transaction id")) return "transaction_id";
    if (c.includes("date") || c.includes("created")) return "transaction_date";
    if (c.includes("sku") || c.includes("product id") || c.includes("item id") || c.includes("article")) return "sku_id";
    if (c.includes("product name") || c.includes("item name") || c.includes("title") || c.includes("description")) return "product_name";
    if (c.includes("brand")) return "brand";
    if (c.includes("category") || c.includes("dept")) return "sku_category";
    if (c.includes("manufacturer") || c.includes("vendor") || c.includes("seller")) return "manufacturer_name";
    if (c.includes("state") || c.includes("region") || c.includes("zone")) return "region";
    if (c.includes("channel") || c.includes("platform")) return "channel_type";
    if (c.includes("qty") || c.includes("quantity") || c.includes("units")) return "units_sold";
    if (c.includes("mrp") || c.includes("market price") || c.includes("retail price")) return "mrp";
    if (c.includes("invoice") || c.includes("amount") || c.includes("total") || c.includes("value") || c.includes("price")) return "invoice_value_incl_gst";
    return null;
  };

  const mappings = rawColumns.map(col => ({ raw: col, target: autoMap(col) }));
  const mappedCount   = mappings.filter(m => m.target).length;
  const unmappedCount = mappings.filter(m => !m.target).length;

  const colorMap = {
    transaction_id: "#2563EB", transaction_date: "#7C3AED",
    sku_id: "#DC2626", product_name: "#059669",
    brand: "#D97706", sku_category: "#0891B2",
    manufacturer_name: "#7C3AED", region: "#DB2777",
    channel_type: "#EA580C", units_sold: "#16A34A",
    mrp: "#9333EA", invoice_value_incl_gst: "#2563EB",
  };

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: 20 }}>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start" }}>
        <div>
          <h2 style={{ fontFamily: FONTS.display, fontSize: 22, fontWeight: 700, color: COLORS.text, margin: 0 }}>Mapping & Cleaning</h2>
          <p style={{ fontFamily: FONTS.body, fontSize: 13, color: COLORS.textSecondary, margin: "6px 0 0" }}>
            Select an uploaded file to see its raw columns and how they map to the universal schema.
          </p>
        </div>
        {fileDetails && (
          <div style={{ display: "flex", gap: 8 }}>
            <Badge color={COLORS.success} bg={COLORS.successLight}>{mappedCount} Auto-Mapped</Badge>
            {unmappedCount > 0 && <Badge color={COLORS.amber} bg={COLORS.amberLight}>{unmappedCount} Unmatched</Badge>}
          </div>
        )}
      </div>

      {/* Platform-grouped file selector */}
      <div style={{ background: COLORS.surface, borderRadius: 12, border: `1px solid ${COLORS.border}`, padding: "16px 18px" }}>
        <label style={{ fontFamily: FONTS.body, fontSize: 12, fontWeight: 600, color: COLORS.textMuted, textTransform: "uppercase", letterSpacing: "0.05em", display: "block", marginBottom: 12 }}>
          Select a file to inspect
        </label>
        {uploadedFiles.length === 0 ? (
          <div style={{ fontFamily: FONTS.body, fontSize: 13, color: COLORS.textMuted, padding: "8px 0" }}>
            No files uploaded yet. Go to Ingestion tab to upload files first.
          </div>
        ) : (() => {
          // Detect platform from filename (mirrors DataIngestionHub logic)
          const detectPlatform = (filename) => {
            const n = (filename || "").toLowerCase();
            if (n.includes("shopify"))    return "Shopify";
            if (n.includes("amazon"))     return "Amazon";
            if (n.includes("myntra"))     return "Myntra";
            if (n.includes("woocommerce") || n.includes("woo")) return "WooCommerce";
            if (n.includes("bigcommerce") || n.includes("big")) return "BigCommerce";
            if (n.includes("flipkart"))   return "Flipkart";
            if (n.includes("meesho"))     return "Meesho";
            return "Manual / CSV";
          };
          const platformConfig = {
            "Shopify":     { color: "#1DB954", bg: "#D1FAE5", icon: "🛍️" },
            "Amazon":      { color: "#FF9900", bg: "#FEF3C7", icon: "📦" },
            "Myntra":      { color: "#E91E8C", bg: "#FCE7F3", icon: "👗" },
            "WooCommerce": { color: "#7F54B3", bg: "#EDE9FE", icon: "🛒" },
            "BigCommerce": { color: "#34495E", bg: "#F1F5F9", icon: "🏪" },
            "Flipkart":    { color: "#2874F0", bg: "#DBEAFE", icon: "🔷" },
            "Meesho":      { color: "#F43F5E", bg: "#FEE2E2", icon: "🏷️" },
            "Manual / CSV":{ color: "#64748B", bg: "#F1F5F9", icon: "📄" },
          };
          // Group files by platform
          const grouped = uploadedFiles.reduce((acc, f) => {
            const key = detectPlatform(f.filename);
            if (!acc[key]) acc[key] = [];
            acc[key].push(f);
            return acc;
          }, {});
          return (
            <div style={{ display: "grid", gridTemplateColumns: "repeat(3, 1fr)", gap: 12 }}>
              {Object.entries(grouped).map(([platform, files]) => {
                const cfg = platformConfig[platform] || { color: COLORS.primary, bg: COLORS.primaryLight, icon: "📁" };
                const isPlatformSelected = files.some(f => f.id === selectedFileId);
                return (
                  <div key={platform} style={{
                    borderRadius: 12, border: `1px solid ${isPlatformSelected ? cfg.color + "60" : COLORS.border}`,
                    overflow: "hidden",
                    boxShadow: isPlatformSelected ? `0 0 0 2px ${cfg.color}20` : "none",
                    transition: "all 0.15s",
                  }}>
                    {/* Platform header */}
                    <div style={{
                      display: "flex", alignItems: "center", gap: 8,
                      padding: "10px 14px", background: cfg.bg,
                      borderBottom: `1px solid ${cfg.color}20`,
                    }}>
                      <span style={{ fontSize: 16 }}>{cfg.icon}</span>
                      <span style={{ fontFamily: FONTS.display, fontSize: 13, fontWeight: 700, color: cfg.color }}>{platform}</span>
                      <span style={{
                        marginLeft: "auto", padding: "1px 7px", borderRadius: 10,
                        background: `${cfg.color}20`, fontFamily: FONTS.mono,
                        fontSize: 10, fontWeight: 700, color: cfg.color,
                      }}>{files.length}</span>
                    </div>
                    {/* Files in this platform */}
                    <div style={{ background: COLORS.surface }}>
                      {files.map(f => {
                        const { bg: ftBg, color: ftColor } = getFileTypeColor(f.file_type);
                        const isSelected = selectedFileId === f.id;
                        return (
                          <button
                            key={f.id}
                            onClick={() => setSelectedFileId(isSelected ? null : f.id)}
                            style={{
                              display: "flex", alignItems: "center", gap: 8, width: "100%",
                              padding: "9px 14px", cursor: "pointer", textAlign: "left",
                              border: "none", borderBottom: `1px solid ${COLORS.borderLight}`,
                              background: isSelected ? `${cfg.color}12` : "transparent",
                              transition: "background 0.12s",
                            }}
                            onMouseEnter={e => { if (!isSelected) e.currentTarget.style.background = COLORS.surfaceAlt; }}
                            onMouseLeave={e => { if (!isSelected) e.currentTarget.style.background = "transparent"; }}
                          >
                            <span style={{
                              padding: "1px 5px", borderRadius: 4,
                              background: isSelected ? `${cfg.color}25` : ftBg,
                              fontFamily: FONTS.mono, fontSize: 9, fontWeight: 700,
                              color: isSelected ? cfg.color : ftColor, flexShrink: 0,
                            }}>{f.file_type?.toUpperCase()}</span>
                            <span style={{
                              fontFamily: FONTS.body, fontSize: 11, fontWeight: isSelected ? 600 : 400,
                              color: isSelected ? cfg.color : COLORS.text,
                              overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap", flex: 1,
                            }} title={f.filename}>{f.filename}</span>
                            <span style={{ fontFamily: FONTS.mono, fontSize: 10, color: COLORS.textMuted, flexShrink: 0 }}>
                              {(f.row_count || 0).toLocaleString()}r
                            </span>
                          </button>
                        );
                      })}
                    </div>
                  </div>
                );
              })}
            </div>
          );
        })()}
      </div>

      {/* Mapping view */}
      {loadingDetails && (
        <div style={{ padding: 32, textAlign: "center", color: COLORS.textMuted, fontFamily: FONTS.body }}>Loading schema...</div>
      )}

      {fileDetails && !loadingDetails && (
        <div style={{ display: "flex", gap: 16, minHeight: 420 }}>
          {/* Left: Raw Source columns */}
          <div style={{ flex: 1, background: COLORS.surface, borderRadius: 14, border: `1px solid ${COLORS.border}`, overflow: "hidden", display: "flex", flexDirection: "column" }}>
            <div style={{ padding: "14px 18px", borderBottom: `1px solid ${COLORS.border}`, background: COLORS.surfaceAlt, display: "flex", alignItems: "center", justifyContent: "space-between" }}>
              <span style={{ fontFamily: FONTS.display, fontSize: 13, fontWeight: 700, color: COLORS.text }}>Raw Source Data</span>
              <span style={{ fontFamily: FONTS.mono, fontSize: 11, color: COLORS.textMuted }}>{fileDetails.filename}</span>
            </div>
            <div style={{ flex: 1, overflowY: "auto", padding: 4 }}>
              {mappings.map((m, i) => (
                <div key={i} style={{
                  display: "flex", alignItems: "center", gap: 10, padding: "10px 14px",
                  borderBottom: `1px solid ${COLORS.borderLight}`,
                  background: !m.target ? `${COLORS.warningLight}44` : "transparent",
                }}>
                  <Icons.Grip />
                  <div style={{ flex: 1 }}>
                    <span style={{ fontFamily: FONTS.mono, fontSize: 13, fontWeight: 600, color: COLORS.text }}>{m.raw}</span>
                  </div>
                  {m.target ? (
                    <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
                      <span style={{
                        padding: "3px 10px", borderRadius: 6,
                        background: `${colorMap[m.target] || COLORS.primary}15`,
                        border: `1px solid ${colorMap[m.target] || COLORS.primary}40`,
                        fontFamily: FONTS.mono, fontSize: 11, fontWeight: 600,
                        color: colorMap[m.target] || COLORS.primary,
                      }}>{m.target}</span>
                      <Icons.ArrowRight />
                    </div>
                  ) : (
                    <Badge color={COLORS.amber} bg={COLORS.amberLight}>
                      <Icons.Alert /> Unmatched
                    </Badge>
                  )}
                </div>
              ))}
            </div>
          </div>

          {/* Center connector lines */}
          <div style={{ width: 48, display: "flex", alignItems: "center", justifyContent: "center" }}>
            <svg width="48" height="320" viewBox="0 0 48 320">
              {mappings.filter(m => m.target).slice(0, 6).map((m, i) => {
                const col = colorMap[m.target] || COLORS.primary;
                const y1  = 30 + i * 44;
                const y2  = 30 + universalSchema.indexOf(m.target) * 32;
                return (
                  <g key={i}>
                    <line x1="0" y1={y1} x2="48" y2={Math.min(y2, 310)} stroke={col} strokeWidth="1.5" opacity="0.5" />
                    <circle cx="2" cy={y1} r="3" fill={col} />
                    <circle cx="46" cy={Math.min(y2, 310)} r="3" fill={col} />
                  </g>
                );
              })}
            </svg>
          </div>

          {/* Right: Universal Schema */}
          <div style={{ flex: 1, background: COLORS.surface, borderRadius: 14, border: `1px solid ${COLORS.border}`, overflow: "hidden", display: "flex", flexDirection: "column" }}>
            <div style={{ padding: "14px 18px", borderBottom: `1px solid ${COLORS.border}`, background: COLORS.surfaceAlt, display: "flex", alignItems: "center", justifyContent: "space-between" }}>
              <span style={{ fontFamily: FONTS.display, fontSize: 13, fontWeight: 700, color: COLORS.text }}>Universal Schema</span>
              <Badge color={COLORS.primary} bg={COLORS.primaryLight}>Target</Badge>
            </div>
            <div style={{ flex: 1, overflowY: "auto", padding: 4 }}>
              {universalSchema.map((f) => {
                const isMapped = mappings.some(m => m.target === f);
                return (
                  <div key={f} style={{
                    display: "flex", alignItems: "center", gap: 10, padding: "10px 14px",
                    borderBottom: `1px solid ${COLORS.borderLight}`,
                  }}>
                    <span style={{ width: 4, height: 28, borderRadius: 2, background: isMapped ? (colorMap[f] || COLORS.primary) : COLORS.border }} />
                    <span style={{ fontFamily: FONTS.mono, fontSize: 13, fontWeight: 600, color: COLORS.text, flex: 1 }}>{f}</span>
                    {isMapped ? (
                      <span style={{ color: COLORS.success }}><Icons.Check /></span>
                    ) : (
                      <span style={{ fontFamily: FONTS.body, fontSize: 11, color: COLORS.textMuted }}>—</span>
                    )}
                  </div>
                );
              })}
            </div>
          </div>

          {/* HITL sidebar for unmatched */}
          {unmappedCount > 0 && (
            <div style={{
              width: 220, background: COLORS.surface, borderRadius: 14,
              border: `1px solid ${COLORS.warning}60`, overflow: "hidden",
              display: "flex", flexDirection: "column",
              boxShadow: `0 0 0 1px ${COLORS.warning}20, 0 4px 12px rgba(217,119,6,0.08)`,
            }}>
              <div style={{ padding: "12px 16px", background: `${COLORS.warningLight}80`, borderBottom: `1px solid ${COLORS.warning}30`, display: "flex", alignItems: "center", gap: 8 }}>
                <Icons.Alert />
                <span style={{ fontFamily: FONTS.display, fontSize: 12, fontWeight: 700, color: COLORS.amber }}>Human Review</span>
              </div>
              <div style={{ padding: 12, flex: 1, display: "flex", flexDirection: "column", gap: 10 }}>
                <div style={{ fontFamily: FONTS.body, fontSize: 11, color: COLORS.textSecondary, lineHeight: 1.5 }}>
                  These columns could not be auto-matched. Review and map them manually.
                </div>
                {mappings.filter(m => !m.target).map(m => (
                  <div key={m.raw} style={{
                    padding: "8px 10px", borderRadius: 8,
                    border: `1px dashed ${COLORS.warning}80`, background: `${COLORS.warningLight}40`,
                    display: "flex", alignItems: "center", gap: 6,
                  }}>
                    <Icons.Grip />
                    <span style={{ fontFamily: FONTS.mono, fontSize: 11, fontWeight: 600, color: COLORS.text }}>{m.raw}</span>
                  </div>
                ))}
                <div style={{ marginTop: "auto", padding: "10px 0 0", borderTop: `1px solid ${COLORS.border}` }}>
                  <button style={{
                    width: "100%", padding: "8px", borderRadius: 8, border: "none",
                    background: COLORS.primary, color: "white", fontFamily: FONTS.body,
                    fontSize: 12, fontWeight: 600, cursor: "pointer",
                  }}>Confirm Mappings</button>
                </div>
              </div>
            </div>
          )}
        </div>
      )}

      {/* Empty state: no file selected */}
      {!fileDetails && !loadingDetails && uploadedFiles.length > 0 && (
        <div style={{
          padding: "48px 24px", textAlign: "center",
          background: COLORS.surface, borderRadius: 14, border: `1px dashed ${COLORS.border}`,
        }}>
          <div style={{ fontSize: 40, marginBottom: 12 }}>🗺️</div>
          <div style={{ fontFamily: FONTS.display, fontSize: 15, fontWeight: 600, color: COLORS.text }}>Select a file above to see its schema</div>
          <div style={{ fontFamily: FONTS.body, fontSize: 13, color: COLORS.textMuted, marginTop: 4 }}>
            The raw column names will be auto-mapped to the universal schema
          </div>
        </div>
      )}
    </div>
  );
};

  );

  const MiniBar = () => (

      {[{ x: 10, h: 60, color: COLORS.primary }, { x: 58, h: 82, color: "#7C3AED" }, { x: 106, h: 45, color: "#F59E0B" }, { x: 154, h: 30, color: "#059669" }, { x: 202, h: 68, color: "#DC2626" }].map((b, i) => (

                    <path d={w.positive ? "M0,35 Q30,30 60,28 T120,20 T180,12 T200,8 V40 H0 Z" : "M0,10 Q30,12 60,15 T120,22 T180,30 T200,35 V40 H0 Z"} fill={`url(#grad-${w.id})`} />

                </div>
              )}
              {w.type === "chart-donut" && (
                <div style={{ display: "flex", alignItems: "center", gap: 20 }}>
                  <MiniDonut />
                  <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>

                        <span style={{ fontFamily: FONTS.mono, fontSize: 12, fontWeight: 600, color: COLORS.text }}>{d.pct}</span>

                    ))}

                </div>
              )}
              {w.type === "chart-bar" && <MiniBar />}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
